//混合填空题输入框自适应
let mixfillInputArr = document.querySelectorAll('.mixfill-input')
if(mixfillInputArr && mixfillInputArr.length) {
	for(let i=0; i < mixfillInputArr.length; i++) {					                                                mixfillInputArr[i].addEventListener("input",function(){
        this.style.width="0px";//让 scrollWidth 获取最小值，达到回缩的效果
        this.style.width=this.scrollWidth+4+"px";
        });
    }
}
