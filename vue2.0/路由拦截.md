登录时路由拦截

```javascript
router.beforeEach(function (to, from, next) {

 let token = VueCookies.get('tokenWeb')

 let require = to.matched.some(function(item){   // 是否需要登录

  return item.meta.isLogin

 })

 if (VueCookies.get('tokenWeb') && to.name == 'login') {

  Message({

   showClose: true,

   message: '您已登录!',

   type: "warning"

  });

  next(from.path)

 }else{              // 已登录就正常跳转，不做任何拦截

  next()             // 正常跳转到下一页

 }

 if( !token && require ){

  Message({

   showClose: true,

   message: '请登录，登录进入循证注册平台!',

   type: "warning"

  });    // 当未登录，且跳转的页面需要登录后才能操作时，进行路由拦截

  next('/login');

 }else{              // 已登录就正常跳转，不做任何拦截

  next()             // 正常跳转到下一页

 }

})
```
